        async function fetchGemini(prompt, system) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: prompt }] }],
                systemInstruction: { parts: [{ text: system }] }
            };
            try {
                const response = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                const result = await response.json();
                return result.candidates?.[0]?.content?.parts?.[0]?.text || "Respuesta no disponible.";
            } catch (e) { return "Error de conexión con el motor de IA."; }
        }

        async function askGemini() {
            const input = document.getElementById('aiInput');
            const chat = document.getElementById('aiChatContent');
            const userText = input.value.trim();
            if (!userText) return;

            chat.innerHTML += `<div class="bg-blue-600 text-white p-3 rounded-2xl text-right ml-8 text-xs font-black shadow-md border-b-2 border-blue-700">${userText}</div>`;
            input.value = "";
            chat.scrollTop = chat.scrollHeight;

            const sysInstruct = `Eres el experto logístico de Aharoni Imports & Exports. 
            REGLA CRÍTICA: Bajo NINGUNA circunstancia des precios, presupuestos, tarifas numéricas o costos estimados. 
            Si el usuario pregunta por costos, responde que solo nuestras oficinas emiten cotizaciones oficiales debido a la volatilidad del mercado. 
            Indícales que utilicen los botones de WhatsApp para hablar con un agente humano. 
            Enfócate solo en asesoría técnica, tipos de carga y rutas.`;

            const response = await fetchGemini(userText, sysInstruct);
